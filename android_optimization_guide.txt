# إعدادات Android محسنة لتقليل حجم التطبيق

# في android/app/build.gradle أضف هذه الإعدادات:

android {
    # تفعيل ProGuard لتقليل الحجم
    buildTypes {
        release {
            shrinkResources true
            minifyEnabled true
            useProguard true
            proguardFiles getDefaultProguardFile('proguard-android-optimize.txt'), 'proguard-rules.pro'
        }
    }
    
    # تحسين ABI
    splits {
        abi {
            enable true
            reset()
            include 'arm64-v8a', 'armeabi-v7a'
            universalApk false
        }
    }
    
    # ضغط الموارد
    aaptOptions {
        cruncherEnabled = false
    }
}
